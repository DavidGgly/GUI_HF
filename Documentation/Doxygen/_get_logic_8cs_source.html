138  public IList<RacetrackQuery> GetRacetrackQuery()
139  {
140  var queryHelp = from race in this.raceRepo.GetAll()
141  group race by race.Rtrack into grp
142  select new
143  {
144  TrackID = grp.Key,
145  TrackCount = grp.Count(),
146  };
147 
148  var query = from rt in this.racetrackRepo.GetAll()
149  join race in queryHelp on rt.Id equals race.TrackID into grj
150  from subRace in grj
151  orderby subRace.TrackCount descending
152  select new RacetrackQuery()
153  {
154  TrackName = rt.Trackname,
155  SumRaces = subRace.TrackCount,
156  BuiltYear = rt.Builtyear,
157  TrackLength = rt.Tlength,
158  TrackVenue = rt.Tvenue,
